Lesson 2: Control Structures

Learn to control program flow with loops and conditionals. 

# Example: For Loop and If Statement
for i in range(5):
    if i % 2 == 0:
        print(f"{i} is even")
    else:
        print(f"{i} is odd")

Sustainability Project: Carbon Footprint Calculator

Calculate your carbon footprint to promote sustainable living. 

# Carbon Footprint Calculator
def calculate_carbon_footprint(km_driven, electricity_kwh):
    car_emission = km_driven * 0.12  # kg CO2 per km
    electricity_emission = electricity_kwh * 0.4  # kg CO2 per kWh
    total = car_emission + electricity_emission
    return total

km = float(input("Enter kilometers driven today: "))
kwh = float(input("Enter electricity usage (kWh) today: "))
footprint = calculate_carbon_footprint(km, kwh)
print(f"Your carbon footprint: {footprint} kg CO2")
